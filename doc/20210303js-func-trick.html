短小而实用的JS代码片段

# 浮点数取整

  const x = 123.4545
  Math.floor(x); // 123

  // 以下方法只适用于32位以下整数
  x >> 0; // 123
  ~~x; // 123
  x | 0; // 123

  // 关于负数
  Math.floor(-12.53); // -13
  -12.53 | 0; // -12

# 六位数字验证码

  ('000000' + Math.floor(Math.random() * 999999)).slice(-6)

# 16进制颜色代码生成

  '#' + ('000000' + (Math.random() * 0x1000000<<0)).toString(16).slice(-6)
